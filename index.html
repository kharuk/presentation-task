Scope

Scope

is a logical boundaries, that determines the accessibility of variables, objects, and functions from different parts of the code

Scope

  • Global
  • Local

Global scope


var global = 'I am global variable!';
function saySomething() {
  console.log(global);
}
console.log(global); // 'I am global variable!'
saySomething(); // 'I am global variable!'

Declaration with var


var thing = 'something';
var thing = 'something else';
console.log(thing); //'something else'

Declaration with let/const


// Don't do this!
let thing = 'something';
let thing = 'something else'; // Error

Local Scope

Variables are usable only in a specific part of the code

  • function scope
  • block scope(ES2015+)

Function scope

Variables are accessible only within the function


function localScopeExample(){
  // LOCAL SCOPE
  var animal = 'Cat';
  console.log(animal); // Cat
}
// GLOBAL SCOPE
console.log(animal); // ReferenceError: animal is not defined

Usage the same variable name in different functions


function firstFunction(){
  var animal = 'Cat';
  console.log(animal); // Cat
}

function secondFunction(){
  var animal = 'Dog';
  console.log(animal); // Dog
}

Functions do not have access to each other's scopes


function firstFunction(){
  var numberOfFunction = 1;
  var animal = 'Cat';
  console.log(animal); // Cat
}

function secondFunction(){
  var animal = 'Dog';
  console.log(animal); // Dog
  firstFunction();
  console.log(numberOfFunction); //Error
}

Nested scopes

the ability of an inner function to access the scope of an outer function


function outerFunction () {
  const outer = `I'm the outer function!`

  function innerFunction() {
    const inner = `I'm the inner function!`
    console.log(outer) // I'm the outer function!
  }

  console.log(inner) // Error, inner is not defined
}

Scope shadowing

the priority of the local variable over the global


var scope = "global"; //Declare a global variable
function checkscope() {
  var scope = "local"; //Declare a local variable with the same name
  return scope; //Return the local value, not the global one
}
checkscope() //"local"

Hoisting

movement variable declarations to the beginning of their scopes.

Function hoisting


saySomething();

function saySomething () {
  console.log('Hello!');
}
              

function saySomething () {
  console.log('Hello!');
}

saySomething();
								

saySomething(); // Error, saySomething is not defined
const saySomething = function () {
  console.log('Hello!');
};

Hoisting of variables


var scope = "global";
function f() {
  console.log(scope);
  var scope = "local";
  console.log(scope);
}
            

var scope = "global";
function f() {
  console.log(scope);  // Prints "undefined", not "global"
  var scope = "local"; // Variable initialized here, but defined everywhere
  console.log(scope);  // Prints "local"
}

Hoisted of variable declaration, but no initialization


function f() {
  var scope;
  console.log(scope); //"undefined"
  scope = "local";
  console.log(scope); //"local"
}
